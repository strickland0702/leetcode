#### [739. 每日温度](https://leetcode-cn.com/problems/daily-temperatures/)

```python
class Solution:
    def dailyTemperatures(self, temperatures: List[int]) -> List[int]:

        # 单调栈
        # 找右侧最近的比arr[i]大的值

        n = len(temperatures)

        results = n * [0]
        stack = []

        for i in range(len(temperatures)):
            while stack and temperatures[stack[-1]] < temperatures[i]:
                j = stack.pop()
                results[j] = i - j

            stack.append(i)

        while stack:
            j = stack.pop()
            results[j] = 0

        return results

```

#### [84. 柱状图中最大的矩形](https://leetcode-cn.com/problems/largest-rectangle-in-histogram/)

```python
class Solution:
    def largestRectangleArea(self, heights: List[int]) -> int:

        if not heights:
            return 0

        maxarea = 0

        stack = []

        for i in range(len(heights)):
            while stack and heights[i] <= heights[stack[-1]]:
                j = stack.pop()
                left = stack[-1] if stack else -1
                right = i
                area = (right - left - 1) * heights[j]
                maxarea = max(area, maxarea)

            stack.append(i)

        while stack:
            j = stack.pop()
            left = stack[-1] if stack else -1
            right = len(heights)
            area = (right - left - 1) * heights[j]
            maxarea = max(area, maxarea)

        return maxarea
```

#### [85. 最大矩形](https://leetcode-cn.com/problems/maximal-rectangle/)

```python
class Solution:
    def maximalRectangle(self, matrix: List[List[str]]) -> int:

        # MaxArea方法与84题完全相同
        def MaxArea(heights):
            maxarea = 0
            stack = []

            for i in range(len(heights)):
                while stack and heights[i] <= heights[stack[-1]]:
                    j = stack.pop()
                    left = stack[-1] if stack else -1
                    right = i
                    area = (right - left - 1) * heights[j]
                    maxarea = max(area, maxarea)

                stack.append(i)

            while stack:
                j = stack.pop()
                left = stack[-1] if stack else -1
                right = len(heights)
                area = (right - left - 1) * heights[j]
                maxarea = max(area, maxarea)

            return maxarea

        maxarea = 0
        rows = len(matrix)
        cols = len(matrix[0])
        heights = cols * [0]

        # 考虑以矩阵每一层为底向上形成的矩阵
        for i in range(rows):           
            for j in range(cols):
                # 如果有0，视为截断
                heights[j] = 0 if matrix[i][j] == "0" else heights[j] + 1

            maxarea = max(MaxArea(heights), maxarea)

        return maxarea

```

#### [496. 下一个更大元素 I](https://leetcode-cn.com/problems/next-greater-element-i/)

```python
class Solution:
    def nextGreaterElement(self, nums1: List[int], nums2: List[int]) -> List[int]:

        results = []
        num_to_index = {}
        for index, num2 in enumerate(nums2):
            num_to_index[num2] = index

        for num1 in nums1:
            index = num_to_index[num1]

            flag = False
            for i in range(index + 1, len(nums2)):
                if nums2[i] > num1:
                    results.append(nums2[i])
                    flag = True
                    break

            if not flag:
                results.append(-1)

        return results
```

#### [907. 子数组的最小值之和](https://leetcode-cn.com/problems/sum-of-subarray-minimums/)

```python
class Solution:
    def sumSubarrayMins(self, A: List[int]) -> int:
        MOD = 10**9 + 7
        N = len(A)

        stack = []
        prev = [None] * N
        next_ = [None] * N

        # 维护栈底到栈顶 小到大的栈时
        # 待进入 <= 栈顶，左边严格小，右边找到相等元素
        # 待进入 < 栈顶，左边找到相等元素，右边严格小

        # 维护栈底到栈顶 大到小的栈时
        # 带进入 >= 栈顶，左边严格大，右边找到相等元素
        # 带进入 > 栈顶，左边找到相等元素，右边严格大

        for i in range(N):
            while stack and A[i] <= A[stack[-1]]:
                j = stack.pop()
                prev[j] = stack[-1] if stack else -1
                next_[j] = i
                stack.append(i)

                while stack:
                    j = stack.pop()
                    prev[j] = stack[-1] if stack else -1
                    next_[j] = N
           
       	# 另一种单调栈的写法
        # prev has i* - 1 in increasing order of A[i* - 1]
        # where i* is the answer to query j
        stack = []
        prev = [None] * N
        for i in range(N):
            while stack and A[i] <= A[stack[-1]]:
                stack.pop()
            prev[i] = stack[-1] if stack else -1
            stack.append(i)

        # next has k* + 1 in increasing order of A[k* + 1]
        # where k* is the answer to query j
        stack = []
        next_ = [None] * N
        for k in range(N-1, -1, -1):
            while stack and A[k] < A[stack[-1]]:
                stack.pop()
            next_[k] = stack[-1] if stack else N
            stack.append(k)

            sum_ = 0
            for i in range(N):
                sum_ += (i - prev[i]) * (next_[i] - i) * A[i]
            
            return sum_ % MOD
```
