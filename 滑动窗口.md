```python
        # return False
             
        # 滑动窗口

        # 考虑长度为k+1的滑动窗口，其中任意两个元素的距离<=k
        # 如果存在两个元素相等，则成立
        
        window = set()

        for i, num in enumerate(nums):
            if i > k:
                window.remove(nums[i-k-1])

            if num in window:
                return True

            window.add(num)

        return False
```

#### [3. 无重复字符的最长子串](https://leetcode-cn.com/problems/longest-substring-without-repeating-characters/)

```python
class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:

        # 暴力法
        # chars = list(s)
        # if len(chars) == 0:
        #     return 0

        # n = len(chars)
        # max_size = 1

        # for i, char in enumerate(chars):
        #     char_set = set()
        #     char_set.add(chars[i])

        #     for j in range(i+1, n):
        #         if chars[j] in char_set:
        #             break

        #         char_set.add(chars[j])

        #     max_size = max(max_size, len(char_set))

        # return max_size

        # 滑动窗口
        # 设由第i个字母开头的最长无重复字串的末尾下标为r_i
        # 则由第i+1字母开头的最长无重复字串的末尾下标一定大于等于r_i（在前面一段肯定不重复）
        # 相当于左边界往右移动，有边界也往右移动，每次有边界停在第一次出现重复字符的位置，经典的滑动窗口

        window = set()
        right = 0
        max_length = 0
        n = len(s)

        for left in range(n):
            if left != 0:
                window.remove(s[left-1])

            while right < n and s[right] not in window:
                window.add(s[right])
                right += 1
                
            # 此时right指针停在了第一个出现重复字符的位置
            # 无重复子串为s[left:right], 长度为right-left

            max_length = max(max_length, right - left)

        return max_length
```
